Zod -- 
with zod, we create schemas which we can then infer as types as well, without it, need to manually create
types alongside schemas

-- example
UserSchema = z.object({
    name: z.string().min(),
    contact: z.number().min()
})

type User --- NOT NEEDED WITH ZOD
-- rather, do this --
type User = z.infer<typeof UserSchema>

-- Validating
UserSchema.parse -> a function on every schema object used to parse some input and returns the typed object.

-- Primitives
z.string();
z.number();
z.boolean();
z.date();
z.undefined();
z.null();

-- Objects
UserSchema = z.object({
    id: z.string(),
    email: z.string().email()  -- "NOT USING BRACKETS AFTERWARDS"
    created_at: z.date,
    isActive: z.boolean,
    age: z.number.positive
})

-- Arrays
-- Usage => z.array(type) 
const NumberedArraySchema = z.array(z.number())

-- With constraints
z.array(z.number()).min().max() => min and max number of items to have in an array

-- Default Vals
const theme = z.string().default("light)

-- No theme val nneded when creating or parsing this schema

-- Additional Validations on string
z.object({
    password: z.string().min(6, {error}).
    max().
    regex(/[A-Z]/, {error}).
    regex(/[0-9]/, {error}), 
    ... => more regex expressions ...
})

-- some more string useful zod utilities 
trim, lowercase, uppercase


-- Number Validations
z.number().min().max()

PriceSchema = z.number.nonnegative -- can be 0
              .finite


-- Enums and Literals
z.enum([...enum_vals])

say you have a native tsc enum, you can convert that to a zod enum by passing that to the z.nativeEnum
utility

to create a literal, i.e a enum val use z.literal()

-- Unions and DiscriminatedUnions

const StrignOrNumber = z.union([z.number(), z.string()])
type StrignOrNumber = z.infer<typeof StrignOrNumber> => string | number
