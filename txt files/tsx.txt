Extracting a return type of a function

type ReturnType<T> = T extends (...args: any[]) => infer R ? R : never

function getUser() {
    return { id: "", name: "" }
}

type UserReturn = ReturnType<typeof getUser>

a better usage of recurive types -
type DeepRead<T> = {
    [K in keyof T]?: T[K] extends object 
    ? DeepRead<T[K]> : T[K]
}

type Config = {
    server: {
        host: string;
        port: number;
    }
}

type ReadonlyConfig = DeepRead<Config>
